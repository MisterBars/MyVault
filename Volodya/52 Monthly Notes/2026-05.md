## Закрытые квесты

```dataviewjs
const { DateTime } = luxon;
const monthNote = dv.current().file.name;
const monthBase = DateTime.fromFormat(monthNote, "yyyy-MM");
const from = monthBase.startOf("month");
const to = monthBase.endOf("month");
const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };

const tasks = dv.pages()
  .where(x =>
    x.type === "task" &&
    x.status === "done" &&
    x.done_date &&
    dv.date(x.done_date).ts >= from.ts &&
    dv.date(x.done_date).ts <= to.ts &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

const modules = dv.pages()
  .where(x =>
    (x.type === "module" || x.type === "class" || x.type === "form") &&
    x.status === "done" &&
    x.done_date &&
    dv.date(x.done_date).ts >= from.ts &&
    dv.date(x.done_date).ts <= to.ts &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

const bySkill = {};

for (const t of tasks) {
  const skill = t.skill || "без навыка";
  const xp = t.xp || rewards[t.task_type || "simple"] || 10;
  bySkill[skill] = (bySkill[skill] || 0) + xp;
}

for (const m of modules) {
  const skill = m.skill || "без навыка";
  const xp = m.reward_xp || 0;
  bySkill[skill] = (bySkill[skill] || 0) + xp;
}

const rows = Object.entries(bySkill).sort((a, b) => b[1] - a[1]);
const total = rows.reduce((s, [, x]) => s + x, 0);

if (rows.length === 0) {
  dv.el("div", "Нет завершённых задач/модулей за этот месяц");
} else {
  dv.table(["Навык", "XP за месяц"], rows);
  dv.el("b", "Итого за месяц: " + total + " XP");
}
```

## Достижения месяца

```dataviewjs
const { DateTime } = luxon;
const monthNote = dv.current().file.name;
const monthBase = DateTime.fromFormat(monthNote, "yyyy-MM");
const from = monthBase.startOf("month");
const to = monthBase.endOf("month");

const rows = dv.pages()
  .where(x =>
    x.type === "achievement" &&
    x.date &&
    dv.date(x.date).ts >= from.ts &&
    dv.date(x.date).ts <= to.ts &&
    !x.file.path.includes("90 Templates")
  )
  .sort(x => x.date, "desc")
  .map(x => [x.file.link, x.date, x.achievement_xp || 0]);

if (rows.length === 0) {
  dv.paragraph("Нет достижений за этот месяц.");
} else {
  dv.table(["Достижение", "Дата", "XP"], rows);
}
```
