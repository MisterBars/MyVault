function clear(el) {
  while (el.firstChild) el.removeChild(el.firstChild);

function makeCard(parent, extra = "") {
  const el = document.createElement("div");
  el.style.cssText = `

    border-radius:10px;
    border:1px solid var(--background-modifier-border);

  parent.appendChild(el);
  return el;

function makeTitle(parent, text, level = 2) {

function makeTable(parent, headers, rows) {
  const tableWrap = document.createElement("div");
  tableWrap.style.cssText = "overflow-x:auto;";
  parent.appendChild(tableWrap);

  const table = document.createElement("table");
  table.style.cssText = `
    width:100%;
    border-collapse:collapse;
    font-size:0.95em;
  `;
  tableWrap.appendChild(table);

  const thead = document.createElement("thead");
  table.appendChild(thead);

  const headRow = document.createElement("tr");
  thead.appendChild(headRow);

  for (const h of headers) {
    const th = document.createElement("th");
    th.textContent = h;
    th.style.cssText = `
      text-align:left;
      padding:8px;
      border-bottom:1px solid var(--background-modifier-border);
      white-space:nowrap;
    `;
    headRow.appendChild(th);
  }

  const tbody = document.createElement("tbody");
  table.appendChild(tbody);

  for (const row of rows) {
    const tr = document.createElement("tr");
    tbody.appendChild(tr);

    for (const cell of row) {
      const td = document.createElement("td");
      td.style.cssText = `
        padding:8px;
        border-bottom:1px solid var(--background-modifier-border-hover);
        vertical-align:top;
      `;

      if (cell instanceof HTMLElement) td.appendChild(cell);
      else td.textContent = cell == null ? "" : String(cell);

      tr.appendChild(td);
    }
  }

  return table;
}

function makeInternalLink(linkObj) {
  const a = document.createElement("a");
  a.className = "internal-link";
  a.href = linkObj?.path || "";
  a.textContent = linkObj?.display ?? (linkObj?.path ? linkObj.path.split("/").pop().replace(".md", "") : "");
  a.style.cursor = "pointer";

  a.addEventListener("click", async (evt) => {
    evt.preventDefault();
    if (linkObj?.path) {
      await app.workspace.openLinkText(linkObj.path, dv.current().file.path);
    }
  });

  return a;
}

function levelFromXP(xp) {
  return Math.floor(Math.sqrt(xp / 100)) + 1;
}

function xpForLevel(level) {
  return Math.pow(level - 1, 2) * 100;
}

function smartMax(arr) {
  const max = Math.max(0, ...arr);
  if (max <= 5) return 5;
  if (max <= 10) return 10;
  if (max <= 25) return 25;
  if (max <= 50) return 50;
  if (max <= 100) return 100;
  if (max <= 250) return 250;
  return Math.ceil(max * 1.15);
}

function formatPeriodLabel() {
  if (state.period === "all") return "Всё время";
  if (state.period === "year") return "Год: " + state.baseDate.toFormat("yyyy");
  if (state.period === "month") return "Месяц: " + state.baseDate.toFormat("LLLL yyyy");
  if (state.period === "week") {
    const start = state.baseDate.startOf("week");
    const end = start.plus({ days: 6 });
    return "Неделя: " + start.toFormat("dd.MM.yyyy") + " → " + end.toFormat("dd.MM.yyyy");
  }
  return "";
}

function resetToToday() {
  const today = dv.date("today");
  if (state.period === "year") state.baseDate = today.startOf("year");
  else if (state.period === "month") state.baseDate = today.startOf("month");
  else if (state.period === "week") state.baseDate = today.startOf("week");
  else state.baseDate = today;
}

function movePeriod(step) {
  if (state.period === "all") return;
  if (state.period === "year") state.baseDate = state.baseDate.plus({ years: step }).startOf("year");
  if (state.period === "month") state.baseDate = state.baseDate.plus({ months: step }).startOf("month");
  if (state.period === "week") state.baseDate = state.baseDate.plus({ days: 7 * step }).startOf("week");
}

function inPeriod(dateObj) {
  if (state.period === "all") return true;

  if (state.period === "year") {
    return dateObj.year === state.baseDate.year;
  }

  if (state.period === "month") {
    return dateObj.year === state.baseDate.year && dateObj.month === state.baseDate.month;
  }

  if (state.period === "week") {
    const start = state.baseDate.startOf("week");
    const end = start.plus({ days: 6 });
    return dateObj >= start && dateObj <= end;
  }

  return true;
}

function bucketKey(d) {
  if (state.period === "all") return d.toFormat("yyyy-MM");
  if (state.period === "year") return d.toFormat("yyyy-MM");
  if (state.period === "month") return d.toFormat("yyyy-MM-dd");
  if (state.period === "week") return d.toFormat("yyyy-MM-dd");
  return d.toFormat("yyyy-MM");
}

function collectEvents() {
  const events = [];

  for (const t of dv.pages().where(x =>
    x.type === "task" &&
    x.status === "done" &&
    x.done_date &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )) {
    const d = dv.date(t.done_date);
    events.push({
      date: d,
      dateStr: d.toFormat("yyyy-MM-dd"),
      type: "task",
      subtype: t.task_type || "simple",
      link: t.file.link,
      skill: t.skill || "без навыка",
      xp: t.xp || rewards[t.task_type || "simple"] || 10
    });
  }

  for (const m of dv.pages().where(x =>
    (x.type === "module" || x.type === "class" || x.type === "form") &&
    x.status === "done" &&
    x.done_date &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )) {
    const d = dv.date(m.done_date);
    events.push({
      date: d,
      dateStr: d.toFormat("yyyy-MM-dd"),
      type: "module",
      subtype: m.type,
      link: m.file.link,
      skill: m.skill || "без навыка",
      xp: m.reward_xp || 0
    });
  }

  for (const p of dv.pages().where(x =>
    x.type === "project" &&
    x.status === "done" &&
    x.done_date &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )) {
    const d = dv.date(p.done_date);
    events.push({
      date: d,
      dateStr: d.toFormat("yyyy-MM-dd"),
      type: "project",
      subtype: "project",
      link: p.file.link,
      skill: "project",
      xp: p.reward_xp || 0
    });
  }

  for (const a of dv.pages().where(x =>
    x.type === "achievement" &&
    x.date &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )) {
    const d = dv.date(a.date);
    events.push({
      date: d,
      dateStr: d.toFormat("yyyy-MM-dd"),
      type: "achievement",
      subtype: a.category || "achievement",
      link: a.file.link,
      skill: a.skill || "achievement",
      xp: a.achievement_xp || 0
    });
  }

  events.sort((a, b) => a.date.ts - b.date.ts);
  return events;
}

function renderProfileHeader(events) {
  clear(profileRoot);

  const p = dv.current();
  const totalXPAll = events.reduce((s, e) => s + e.xp, 0);
  const level = levelFromXP(totalXPAll);
  const levelStartXP = xpForLevel(level);
  const nextLevelXP = xpForLevel(level + 1);
  const currentLevelProgress = totalXPAll - levelStartXP;
  const neededForNext = nextLevelXP - levelStartXP;
  const remainToNext = Math.max(0, nextLevelXP - totalXPAll);
  const percent = neededForNext <= 0 ? 100 : Math.max(0, Math.min(100, (currentLevelProgress / neededForNext) * 100));

  const card = makeCard(profileRoot, "padding:16px;");

  const title = document.createElement("div");
  title.textContent = "⚔️ " + (p.name || p.file.name || "Персонаж");
  title.style.cssText = "font-size:1.6em;font-weight:800;margin-bottom:4px;color:#7bd389;";
  card.appendChild(title);

  const subtitle = document.createElement("div");
  subtitle.textContent = p.class || "";
  subtitle.style.cssText = "color:#82aaff;margin-bottom:14px;font-weight:600;";
  card.appendChild(subtitle);

  const statsRow = document.createElement("div");
  statsRow.style.cssText = "display:grid;grid-template-columns:repeat(auto-fit,minmax(180px,1fr));gap:10px;margin-bottom:12px;";
  card.appendChild(statsRow);

  const stat1 = makeCard(statsRow, "margin-bottom:0;");
  makeText(stat1, "Уровень", "color:var(--text-muted);");
  makeText(stat1, String(level), "font-size:1.7em;font-weight:800;color:#82aaff;");

  const stat2 = makeCard(statsRow, "margin-bottom:0;");
  makeText(stat2, "Общий XP", "color:var(--text-muted);");
  makeText(stat2, String(totalXPAll), "font-size:1.7em;font-weight:800;color:#f7c46c;");

  const stat3 = makeCard(statsRow, "margin-bottom:0;");
  makeText(stat3, "До следующего уровня", "color:var(--text-muted);");
  makeText(stat3, String(remainToNext), "font-size:1.7em;font-weight:800;color:#c792ea;");

  const stat4 = makeCard(statsRow, "margin-bottom:0;");
  makeText(stat4, "Всего событий", "color:var(--text-muted);");
  makeText(stat4, String(events.length), "font-size:1.7em;font-weight:800;color:#89ddff;");

  const progressLabel = document.createElement("div");
  progressLabel.textContent = `Прогресс уровня: ${Math.round(percent)}% (${totalXPAll} / ${nextLevelXP} XP)`;
  progressLabel.style.cssText = "margin-bottom:6px;color:var(--text-normal);";

  progressOuter.style.cssText = `

    height:14px;
    border-radius:999px;
    background:var(--background-primary);
    overflow:hidden;
    border:1px solid var(--background-modifier-border);

    width:${percent}%;
    height:100%;
    background:linear-gradient(90deg, #7bd389, #82aaff);
  `;
  progressOuter.appendChild(progressInner);

function renderControls(events) {
  clear(controls);

  makeButton(controls, "Всё время", () => {
    state.period = "all";
    renderAll();
  }, state.period === "all");

  makeButton(controls, "Год", () => {
    state.period = "year";
    state.baseDate = dv.date("today").startOf("year");
    renderAll();

  makeButton(controls, "Месяц", () => {
    state.period = "month";
    state.baseDate = dv.date("today").startOf("month");
    renderAll();
  }, state.period === "month");

  makeButton(controls, "Неделя", () => {
    state.period = "week";
    state.baseDate = dv.date("today").startOf("week");
    renderAll();
  }, state.period === "week");

  makeButton(controls, "←", () => {
    movePeriod(-1);
    renderAll();
  });

  makeButton(controls, "Сегодня", () => {
    resetToToday();
    renderAll();
  });

  makeButton(controls, "→", () => {
    movePeriod(1);
    renderAll();
  });

  const label = document.createElement("div");
  label.textContent = formatPeriodLabel();
  label.style.cssText = "margin-left:6px;font-weight:700;";
  controls.appendChild(label);

  const spacer = document.createElement("div");
  spacer.style.flex = "1";
  controls.appendChild(spacer);

  const skillLabel = document.createElement("span");
  skillLabel.textContent = "Навык:";
  controls.appendChild(skillLabel);

  const select = document.createElement("select");
  select.style.cssText = "padding:6px 8px;border-radius:8px;";

  const skills = Array.from(new Set(events.map(e => e.skill))).sort();

  const allOpt = document.createElement("option");
  allOpt.value = "all";
  allOpt.textContent = "Все";
  select.appendChild(allOpt);

  for (const s of skills) {
    const opt = document.createElement("option");
    opt.value = s;
    opt.textContent = s;
    select.appendChild(opt);
  }

  select.value = state.skill;
  select.addEventListener("change", () => {
    state.skill = select.value;
    renderAll();
  });

  controls.appendChild(select);
}

function renderAll() {
  clear(output);

  const events = collectEvents();
  renderProfileHeader(events);
  renderControls(events);

  const filtered = events.filter(e => {
    if (!inPeriod(e.date)) return false;
    if (state.skill !== "all" && e.skill !== state.skill) return false;
    return true;
  });

  const totalXP = filtered.reduce((s, e) => s + e.xp, 0);
  const totalCount = filtered.length;
  const taskCount = filtered.filter(e => e.type === "task").length;
  const moduleCount = filtered.filter(e => e.type === "module").length;
  const projectCount = filtered.filter(e => e.type === "project").length;
  const achievementCount = filtered.filter(e => e.type === "achievement").length;

  const kpi = makeCard(output);
  makeTitle(kpi, "Срез по выбранному периоду", 3);
  makeText(kpi, "Период: " + formatPeriodLabel());
  makeText(kpi, "Навык: " + (state.skill === "all" ? "Все" : state.skill));
  makeText(kpi, "XP: " + totalXP, "color:#f7c46c;font-weight:700;");
  makeText(kpi, "Событий: " + totalCount, "color:#89ddff;font-weight:700;");
  makeText(kpi, "Задач: " + taskCount + " | Модулей: " + moduleCount + " | Проектов: " + projectCount + " | Достижений: " + achievementCount);

  const buckets = {};
  for (const e of filtered) {
    const key = bucketKey(e.date);
    if (!buckets[key]) buckets[key] = { xp: 0, count: 0, tasks: 0, modules: 0 };
    buckets[key].xp += e.xp;
    buckets[key].count += 1;
    if (e.type === "task") buckets[key].tasks += 1;
    if (e.type === "module") buckets[key].modules += 1;
  }

  const labels = Object.keys(buckets).sort();
  const xpData = labels.map(k => buckets[k].xp);
  const countData = labels.map(k => buckets[k].count);
  const taskData = labels.map(k => buckets[k].tasks);
  const moduleData = labels.map(k => buckets[k].modules);

  let cumulative = 0;
  const cumulativeXP = filtered
    .slice()
    .sort((a, b) => a.date.ts - b.date.ts)
    .reduce((arr, e) => {
      cumulative += e.xp;
      arr.push({ label: bucketKey(e.date), total: cumulative });
      return arr;

  for (const row of cumulativeXP) cumulativeMap[row.label] = row.total;

      data: {
        labels,
        datasets: [
          {
            label: "XP",
            data: xpData,
            borderColor: "#7bd389",
            backgroundColor: "rgba(123,211,137,0.18)",
            yAxisID: "y",
            tension: 0.25,
            fill: true,
            pointRadius: 2
          },
          {
            label: "Завершено",
            data: countData,
            borderColor: "#82aaff",
            backgroundColor: "rgba(130,170,255,0.18)",
            yAxisID: "y1",
            tension: 0.25,
            fill: false,
            pointRadius: 2
          }
        ]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        animation: false,
        interaction: { mode: "index", intersect: false },
        plugins: { legend: { position: "top" } },
        scales: {
          x: {
            ticks: {
              autoSkip: true,
              maxTicksLimit: state.period === "all" ? 12 : 16,
              maxRotation: 0,
              minRotation: 0
            },
            grid: { display: false }
          },
          y: {
            beginAtZero: true,
            suggestedMax: smartMax(xpData),
            ticks: { maxTicksLimit: 6, precision: 0 }
          },
          y1: {
            beginAtZero: true,
            suggestedMax: smartMax(countData),
            position: "right",
            grid: { drawOnChartArea: false },
            ticks: { maxTicksLimit: 6, precision: 0 }

    const card2 = makeCard(output);

      xp: t.xp || rewards[t.task_type || "simple"] || 10,
      date: dv.date(t.done_date)
    });
  }

  for (const m of dv.pages().where(x =>
    (x.type === "module" || x.type === "class" || x.type === "form") &&
    x.status === "done" &&
    x.done_date &&
    x.skill &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )) {
    events.push({
      skill: m.skill,
      xp: m.reward_xp || 0,
      date: dv.date(m.done_date)
    });
  }

  for (const a of dv.pages().where(x =>
    x.type === "achievement" &&
    x.date &&
    x.skill &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )) {
    events.push({
      skill: a.skill,
      xp: a.achievement_xp || 0,
      date: dv.date(a.date)
    });
  }

  return events;
}
