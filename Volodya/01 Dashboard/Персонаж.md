const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };
const skillNotes = dv.pages()
  .where(x =>
    x.type === "skill" &&
    !x.file.path.includes("90 Templates") &&

function levelFromXP(xp) {
  return Math.floor(Math.sqrt(xp / 50)) + 1;

const wrap = dv.el("div", "", {});

for (const s of skillNotes) {
  const id = s.skill_id;
  const tasks = dv.pages()
    .where(x =>
      x.type === "task" &&
      x.status === "done" &&
      x.skill === id &&
      !x.file.path.includes("90 Templates") &&
      !x.file.path.includes("40 Archives")
    );

  const xp = tasks.array().reduce((sum, t) => {
    if (t.xp) return sum + t.xp;
    return sum + (rewards[t.task_type || "simple"] || 10);

  const level = levelFromXP(xp);
  const currentLevelXP = Math.pow(level - 1, 2) * 50;

  const lbl = dv.el("div", "", {container: row});
  lbl.style.cssText = "display:flex;justify-content:space-between;margin-bottom:2px";

  dv.el("span", (s.title || s.file.name) + " · " + level + " уровень", {container: lbl});

  const xpLbl = dv.el("span", xp + " XP", {container: lbl});

  const bw = dv.el("div", "", {container: row});

  const bf = dv.el("div", "", {container: bw});
  bf.style.cssText = "background:" + color + ";width:" + pct + "%;height:8px;border-radius:4px";

## Навыки в таблице

```dataviewjs
const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };
const skillNotes = dv.pages()
  .where(x =>
    x.type === "skill" &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

function levelFromXP(xp) {
  return Math.floor(Math.sqrt(xp / 50)) + 1;
}

const rows = [];

for (const s of skillNotes) {
  const id = s.skill_id;
  const tasks = dv.pages()
    .where(x =>
      x.type === "task" &&
      x.status === "done" &&
      x.skill === id &&
      !x.file.path.includes("90 Templates") &&
      !x.file.path.includes("40 Archives")
    );

  const xp = tasks.array().reduce((sum, t) => {
    if (t.xp) return sum + t.xp;
    return sum + (rewards[t.task_type || "simple"] || 10);
  }, 0);

  rows.push([
    s.file.link,
    tasks.length,
    xp,
    levelFromXP(xp)
  ]);
}

dv.table(["Навык", "Закрыто задач", "XP", "Уровень"], rows);
```

## Достижения

```dataviewjs
const achievements = dv.pages()
  .where(x =>
    x.type === "achievement" &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )
  .sort(x => x.date, "desc");

const wrap = dv.el("div", "", {});
wrap.style.cssText = "padding:12px;background:var(--background-secondary);border-radius:8px;border:1px solid var(--background-modifier-border)";

if (achievements.length === 0) {
  dv.el("div", "Ещё нет достижений. Первое — за то, что начал!", {container: wrap});
} else {
  const categories = { general: "🎖", streak: "🔥", milestone: "🏆", skill: "⚔️" };

  for (const a of achievements) {
    const row = dv.el("div", "", {container: wrap});
    row.style.cssText = "display:flex;justify-content:space-between;align-items:center;padding:4px 0;border-bottom:1px solid var(--background-modifier-border)";

    const icon = categories[a.category] || "🏅";
    const lbl = dv.el("span", icon + " " + a.file.name, {container: row});

    const meta = dv.el("span", "", {container: row});
    meta.style.cssText = "color:var(--text-muted);font-size:0.85em";
    meta.textContent = (a.date || "") + " · +" + (a.achievement_xp || 0) + " XP";
  }

  const total = achievements.array().reduce((s, a) => s + (a.achievement_xp || 0), 0);
  const footer = dv.el("div", "Всего достижений: " + achievements.length + " · " + total + " XP", {container: wrap});
  footer.style.cssText = "margin-top:8px;font-weight:bold";
}
```
