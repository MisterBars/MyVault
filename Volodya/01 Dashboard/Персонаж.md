```dataviewjs
const p = dv.current();
const xp = p.xp_total || 0;
const lvls = Array.of(0,100,200,350,550,800,1100,1500,2000,2600,3300);
let lv = 1;
for (let i = 0; i < lvls.length - 1; i++) {
    if (xp >= lvls[i]) lv = i + 1;
}
const xpA = lvls[lv-1] || 0;
const xpB = lvls[lv] || 9999;
const pct = Math.round(((xp - xpA) / (xpB - xpA)) * 100);
const wrap = dv.el("div", "", {});
wrap.style.cssText = "padding:12px;background:var(--background-secondary);border-radius:8px;border:1px solid var(--background-modifier-border);margin-bottom:8px";
dv.el("div", "🏆 Уровень " + lv + " · " + xp + " XP", {container: wrap});
const bw = dv.el("div", "", {container: wrap});
bw.style.cssText = "background:var(--background-modifier-border);border-radius:4px;height:12px;margin:6px 0";
const bf = dv.el("div", "", {container: bw});
bf.style.cssText = "background:#7c6bea;width:" + pct + "%;height:12px;border-radius:4px";
const inf = dv.el("div", (xp - xpA) + " / " + (xpB - xpA) + " XP до уровня " + (lv+1), {container: wrap});

```

## Навыки

```dataviewjs
const skills = dv.pages('"02 Profile/Skills"').where(p => p.type === "skill");
const wrap = dv.el("div", "", {});
wrap.style.cssText = "padding:12px;background:var(--background-secondary);border-radius:8px;border:1px solid var(--background-modifier-border)";
const clrs = {1:"#888888",2:"#4caf50",3:"#2196f3",4:"#9c27b0",5:"#ff9800"};
for (const s of skills) {
    const xp = s.xp || 0;
    const lv = s.level || 1;
    const pct = Math.min(Math.round((xp / 1000) * 100), 100);
    const color = clrs[Math.min(lv, 5)];
    const row = dv.el("div", "", {container: wrap});
    row.style.cssText = "margin-bottom:10px";
    const lbl = dv.el("div", "", {container: row});
    lbl.style.cssText = "display:flex;justify-content:space-between;margin-bottom:2px";
    dv.el("span", (s.name || s.file.name) + " · Ур." + lv, {container: lbl});
    const xl = dv.el("span", xp + " XP", {container: lbl});
    xl.style.color = "var(--text-muted)";
    const bw = dv.el("div", "", {container: row});
    bw.style.cssText = "background:var(--background-modifier-border);border-radius:4px;height:8px";
    const bf = dv.el("div", "", {container: bw});
    bf.style.cssText = "background:" + color + ";width:" + pct + "%;height:8px;border-radius:4px";
}
```
