# Персонаж
```dataviewjs
const p = dv.current();
const xp = p.xp_total || 0;
const levels =;[1][2]
let level = 1;
for (let i = 0; i < levels.length - 1; i++) {
    if (xp >= levels[i]) level = i + 1;
}
const xpCurrent = levels[level - 1] || 0;
const xpNext = levels[level] || 9999;
const pct = Math.round(((xp - xpCurrent) / (xpNext - xpCurrent)) * 100);
const ranks = ['','Новичок','Ученик','Специалист','Эксперт','Мастер','Архимаг'];
const rank = ranks[Math.min(level, ranks.length - 1)];

const wrap = dv.el('div', '', {});
wrap.style.cssText = 'padding:12px;background:var(--background-secondary);border-radius:8px;border:1px solid var(--background-modifier-border)';

const title = dv.el('div', '⚔️ ' + (p.name || 'Персонаж'), {container: wrap});
title.style.cssText = 'font-size:1.4em;font-weight:bold;margin-bottom:4px';

const sub = dv.el('div', (p.class || '') + ' · ' + rank, {container: wrap});
sub.style.cssText = 'color:var(--text-muted);margin-bottom:12px';

dv.el('div', '🏆 Уровень ' + level + ' · ' + xp + ' XP', {container: wrap});

const barWrap = dv.el('div', '', {container: wrap});
barWrap.style.cssText = 'background:var(--background-modifier-border);border-radius:4px;height:12px;margin:6px 0';

const bar = dv.el('div', '', {container: barWrap});
bar.style.cssText = 'background:#7c6bea;width:' + pct + '%;height:12px;border-radius:4px';

const info = dv.el('div', (xp - xpCurrent) + ' / ' + (xpNext - xpCurrent) + ' XP до уровня ' + (level + 1), {container: wrap});
```

## Навыки

```dataviewjs
const skills = dv.pages('"02 Profile/Skills"').where(p => p.type === "skill");
const maxXp = 1000;

const bars = skills.map(s => {
  const xp = s.xp || 0;
  const lvl = s.level || 1;
  const pct = Math.min(Math.round((xp / maxXp) * 100), 100);
  const colors = {1:'#888',2:'#4caf50',3:'#2196f3',4:'#9c27b0',5:'#ff9800'};
  const color = colors[Math.min(lvl,5)] || '#888';
  return `<div style="margin-bottom:10px;">
    <div style="display:flex; justify-content:space-between; margin-bottom:2px;">
      <span><b>${s.name || s.file.name}</b> · Ур.${lvl}</span>
      <span style="color:var(--text-muted)">${xp} XP</span>
    </div>
    <div style="background:var(--background-modifier-border); border-radius:4px; height:8px;">
      <div style="background:${color}; width:${pct}%; height:8px; border-radius:4px;"></div>
    </div>
  </div>`;
}).join('');

dv.paragraph(`<div style="padding:12px; background:var(--background-secondary); border-radius:8px; border:1px solid var(--background-modifier-border)">${bars}</div>`);
```
