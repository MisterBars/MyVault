# Рус Дмитрий

## Кратко
<p align="justify">Дмитрий Рус — российский писатель-фантаст, один из главных продолжателей жанра LitRPG в русскоязычной литературе. Родился в Киеве, с детства был заядлым читателем, к 13 годам прочитав всю детскую библиотеку и получив доступ во взрослый зал. Получил высшее образование на факультете "Математики и компьютерных наук", некоторое время работал по специальности. Служил в армии дважды в разных странах, получил удостоверение "Снайпер СССР", сохранил любовь к стрельбе.

Долгое время жил за границей — в основном на Востоке и в Азии (около 18 лет), окончательно вернулся и переехал в Москву только в 2014 году. Дебютировал в 2013 году с романом "Играть, чтобы жить. Срыв", который стал настоящим феноменом и заложил основы жанра LitRPG в России. Его культовая серия "Играть, чтобы жить" покоряет читателей глубиной проработки игровой вселенной и головокружительными сюжетными поворотами. Первые две книги цикла были переведены автором на английский язык и выложены на Amazon.</p>

const a = dv.current();

function fmt(v, empty = "—") {
  if (v == null || v === "") return empty;
  if (Array.isArray(v)) return v.length ? v.join(", ") : empty;
  return String(v);
}

function num(v, def = 0) {
  const n = Number(v);
  return Number.isFinite(n) ? n : def;
}

function calcBookXP(book) {
  const pages = num(book.pages);
  if (pages <= 0) return 0;

  const current = Math.min(num(book.current_page), pages);
  const progress = current / pages;
  const baseXP = 100;
  const sizeCoef = num(book.size_coef, 1);
  const difficultyCoef = num(book.difficulty_coef, 1);
  const finishBonus = String(book.status || "") === "done" ? num(book.finish_bonus, 0) : 0;

  return Math.round(progress * baseXP * sizeCoef * difficultyCoef + finishBonus);
}

const doneBooks = books.filter(b => String(b.status || "") === "done");
const ratedBooks = books.filter(b => num(b.rating) > 0);
const avgRating = ratedBooks.length
  ? (ratedBooks.reduce((s, b) => s + num(b.rating), 0) / ratedBooks.length).toFixed(2)
  : "—";
const totalXP = doneBooks.reduce((s, b) => s + calcBookXP(b), 0);

dv.table(

  ]
