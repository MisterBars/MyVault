function calcProgress(book) {
  const pages = num(book.pages);
  const current = Math.min(num(book.current_page), pages);
  if (pages <= 0) return 0;
  return Math.round((current / pages) * 100);
}

function calcXP(book) {
  const pages = num(book.pages);
  if (pages <= 0) return 0;

  const current = Math.min(num(book.current_page), pages);
  const progress = current / pages;
  const baseXP = 100;
  const sizeCoef = num(book.size_coef, 1);
  const difficultyCoef = num(book.difficulty_coef, 1);
  const finishBonus = String(book.status || "") === "done"
    ? num(book.finish_bonus, 0)
    : 0;

  return Math.round(progress * baseXP * sizeCoef * difficultyCoef + finishBonus);
}

## О чём книга
<p align="justify">Архаичная Зона в первые годы после образования. Сталкеры занимаются простым бизнесом — сопровождением туристов, приезжающих из любопытства или в поисках мелких артефактов. Но на сей раз в Зону отправляется не группа праздных любопытствующих, а специальная группа "Табигон" Центра Аномальных Явлений. Цель исследования — выяснить истину о легенде про таинственную Золотую Сферу, обладающую уникальной способностью модулировать реальность, и найти следы предыдущей группы Центра, исчезнувшей при загадочных обстоятельствах. Против людей из "Табигона" — все: официальные лица, охрана, бандиты и сама Зона, подкидывающая новые опасные сюрпризы. Исследователи упорно продолжают миссию и понимают: в Зоне происходит что-то, способное перевернуть весь мир.</p>
