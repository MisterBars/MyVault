function num(v, def = 0) {

function fmt(v, empty = "—") {
  if (v == null || v === "") return empty;

function calcProgress(book) {
  const pages = num(book.pages);
  const current = Math.min(num(book.current_page), pages);
  if (pages <= 0) return 0;
  return Math.round((current / pages) * 100);
}

function calcXP(book) {
  const pages = num(book.pages);
  if (pages <= 0) return 0;

  const current = Math.min(num(book.current_page), pages);
  const progress = current / pages;
  const baseXP = 100;
  const sizeCoef = num(book.size_coef, 1);
  const difficultyCoef = num(book.difficulty_coef, 1);
  const finishBonus = String(book.status || "") === "done"
    ? num(book.finish_bonus, 0)
    : 0;

  return Math.round(progress * baseXP * sizeCoef * difficultyCoef + finishBonus);
}

## О чём книга
<p align="justify">Сын Вадима Малахова идет в Зону, чтобы разыскать своего отца. На своем пути паренек обретает новых неожиданных друзей, раскрывает страшные тайны, при этом неуклонно стремится к своей цели, уничтожая окружающее зло. Что такое мир Зоны, и какие они, ее обитатели, если смотреть на них с противоположной стороны ? Найдет ли сын отца, чем закончится эта смертельная авантюра ? Ответы на эти вопросы содержатся в книге "Красный сигнал" — заключительной части трилогии о Малахове.</p>
