FROM ""
WHERE type = "task" AND status = "done"
  AND deadline = date("<% tp.date.now("YYYY-MM-DD") %>")
  AND !contains(file.path, "90 Templates")
  AND !contains(file.path, "40 Archives")

## XP за сегодня по навыкам

```dataviewjs
const today = "<% tp.date.now("YYYY-MM-DD") %>";
const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };

const tasks = dv.pages()
  .where(x =>
    x.type === "task" &&
    x.status === "done" &&
    x.deadline &&
    dv.date(x.deadline).toFormat("yyyy-MM-dd") === today &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

const bySkill = {};
for (const t of tasks) {
  const skill = t.skill || "без навыка";
  const xp = t.xp || rewards[t.task_type || "simple"] || 10;
  bySkill[skill] = (bySkill[skill] || 0) + xp;
}

const rows = Object.entries(bySkill).sort((a, b) => b - a);[1]
const total = rows.reduce((s, [, x]) => s + x, 0);

if (rows.length === 0) {
  dv.el("div", "Нет завершённых задач с дедлайном сегодня", {});
} else {
  dv.table(["Навык", "XP"], rows);
  dv.el("b", "Итого за день: " + total + " XP", {});
}
```
