## Завершено сегодня

```dataview
const today = "<% tp.date.now("YYYY-MM-DD") %>";
const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };

const doneTasks = dv.pages()
  .where(x =>
    x.type === "task" &&
    x.status === "done" &&
    x.done_date &&
    dv.date(x.done_date).toFormat("yyyy-MM-dd") === today &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )
  .array()
  .map(x => ({
    link: x.file.link,
    kind: "Задача",
    skill: x.skill || "—",
    xp: x.xp || rewards[x.task_type || "simple"] || 10,
    date: x.done_date
  }));

const doneModules = dv.pages()
  .where(x =>
    (x.type === "module" || x.type === "class" || x.type === "form") &&
    x.status === "done" &&
    x.done_date &&
    dv.date(x.done_date).toFormat("yyyy-MM-dd") === today &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )
  .array()
  .map(x => ({
    link: x.file.link,
    kind: x.type === "module" ? "Модуль" : x.type === "class" ? "Класс" : "Форма",
    skill: x.skill || "—",
    xp: x.reward_xp || 0,
    date: x.done_date
  }));

const rows = [...doneTasks, ...doneModules]
  .sort((a, b) => String(b.date).localeCompare(String(a.date)))
  .map(x => [x.link, x.kind, x.skill, x.xp]);

if (rows.length === 0) {
  dv.paragraph("Сегодня ничего не завершено.");
} else {
  dv.table(["Элемент", "Тип", "Навык", "XP"], rows);
}
```

## XP за сегодня по навыкам

```dataviewjs
const today = "<% tp.date.now("YYYY-MM-DD") %>";
const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };

const entries = [];

for (const t of dv.pages().where(x =>
  x.type === "task" &&
  x.status === "done" &&
  x.done_date &&
  dv.date(x.done_date).toFormat("yyyy-MM-dd") === today &&
  !x.file.path.includes("90 Templates") &&
  !x.file.path.includes("40 Archives")
)) {
  entries.push({
    skill: t.skill || "без навыка",
    xp: t.xp || rewards[t.task_type || "simple"] || 10
  });
}

for (const m of dv.pages().where(x =>
  (x.type === "module" || x.type === "class" || x.type === "form") &&
  x.status === "done" &&
  x.done_date &&
  dv.date(x.done_date).toFormat("yyyy-MM-dd") === today &&
  !x.file.path.includes("90 Templates") &&
  !x.file.path.includes("40 Archives")
)) {
  entries.push({
    skill: m.skill || "без навыка",
    xp: m.reward_xp || 0
  });
}

const bySkill = {};
for (const e of entries) bySkill[e.skill] = (bySkill[e.skill] || 0) + e.xp;

const rows = Object.entries(bySkill).sort((a, b) => b[1] - a[1]);
const total = rows.reduce((s, [, x]) => s + x, 0);

if (rows.length === 0) {
  dv.paragraph("Нет завершённых задач/модулей за сегодня.");
} else {
  dv.table(["Навык", "XP"], rows);
  dv.paragraph("**Итого за день:** " + total + " XP");
}
```
