## Завершено сегодня

```dataview
const today = "<% tp.date.now("YYYY-MM-DD") %>";
const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };

// Задачи, завершённые с дедлайном на сегодня
const tasks = dv.pages()
  .where(x =>
    x.type === "task" &&
    x.status === "done" &&
    x.deadline &&
    dv.date(x.deadline).toFormat("yyyy-MM-dd") === today &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

// Модули (модуль/класс/форма), завершённые сегодня по времени изменения
const modules = dv.pages()
  .where(x =>
    (x.type === "module" || x.type === "class" || x.type === "form") &&
    x.status === "done" &&
    x.file.mtime &&
    dv.date(x.file.mtime).toFormat("yyyy-MM-dd") === today &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

const bySkill = {};

// XP из задач
for (const t of tasks) {
  const skill = t.skill || "без навыка";
  const xp = t.xp || rewards[t.task_type || "simple"] || 10;
  bySkill[skill] = (bySkill[skill] || 0) + xp;
}

// XP из модулей
for (const m of modules) {
  const skill = m.skill || "без навыка";
  const xp = m.reward_xp || 0;
  bySkill[skill] = (bySkill[skill] || 0) + xp;
}

const rows = Object.entries(bySkill).sort((a, b) => b[1] - a[1]);
const total = rows.reduce((s, [, x]) => s + x, 0);

if (rows.length === 0) {
  dv.el("div", "Нет завершённых задач/модулей за сегодня", {});
} else {
  dv.table(["Навык", "XP"], rows);
  dv.el("b", "Итого за день: " + total + " XP", {});
}
```
