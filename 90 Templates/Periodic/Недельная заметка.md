# Цели недели

## Завершённые задачи за неделю

```dataviewjs
const weekKey = dv.current().file.name;
const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };

const tasks = dv.pages()
  .where(x =>
    x.type === "task" &&
    x.status === "done" &&
    x.done_date &&
    dv.date(x.done_date).toFormat("kkkk-'W'WW") === weekKey &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

const modules = dv.pages()
  .where(x =>
    (x.type === "module" || x.type === "class" || x.type === "form") &&
    x.status === "done" &&
    x.done_date &&
    dv.date(x.done_date).toFormat("kkkk-'W'WW") === weekKey &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

const bySkill = {};

for (const t of tasks) {
  const skill = t.skill || "без навыка";
  const xp = t.xp || rewards[t.task_type || "simple"] || 10;
  bySkill[skill] = (bySkill[skill] || 0) + xp;
}

for (const m of modules) {
  const skill = m.skill || "без навыка";
  const xp = m.reward_xp || 0;
  bySkill[skill] = (bySkill[skill] || 0) + xp;
}

const rows = Object.entries(bySkill).sort((a, b) => b[1] - a[1]);
const total = rows.reduce((s, [, x]) => s + x, 0);

if (rows.length === 0) {
  dv.paragraph("Нет завершённых задач/модулей за эту неделю.");
} else {
  dv.table(["Навык", "XP за неделю"], rows);
  dv.paragraph("**Итого за неделю:** " + total + " XP");
}
```

## XP за неделю по навыкам

```dataviewjs
const from = dv.current().date_from ? dv.date(dv.current().date_from) : null;
const to = dv.current().date_to ? dv.date(dv.current().date_to) : null;
const rewards = { simple: 10, normal: 20, major: 40, boss: 100 };

const tasks = dv.pages()
  .where(x =>
    x.type === "task" &&
    x.status === "done" &&
    x.deadline &&
    (!from || dv.date(x.deadline) >= from) &&
    (!to || dv.date(x.deadline) <= to) &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  );

const bySkill = {};
for (const t of tasks) {
  const skill = t.skill || "без навыка";
  const xp = t.xp || rewards[t.task_type || "simple"] || 10;
  bySkill[skill] = (bySkill[skill] || 0) + xp;
}

const rows = Object.entries(bySkill).sort((a, b) => b - a);
const total = rows.reduce((s, [, x]) => s + x, 0);

if (rows.length === 0) {
  dv.el("div", "Нет завершённых задач за этот период", {});
} else {
  dv.table(["Навык", "XP за неделю"], rows);
  dv.el("b", "Итого за неделю: " + total + " XP", {});
}
```

## Достижения недели

```dataviewjs
const weekKey = dv.current().file.name;

const pages = dv.pages()
  .where(x =>
    x.type === "achievement" &&
    x.date &&
    !x.file.path.includes("90 Templates") &&
    !x.file.path.includes("40 Archives")
  )
  .array();

const rows = pages
  .map(x => {
    const d = dv.date(x.date);
    if (!d) return null;

    return {
      link: x.file.link,
      date: d,
      xp: x.achievement_xp || 0
    };
  })
  .filter(x => x && x.date.toFormat("kkkk-'W'WW") === weekKey)
  .sort((a, b) => b.date.ts - a.date.ts)
  .map(x => [x.link, x.date.toFormat("yyyy-MM-dd"), x.xp]);

if (rows.length === 0) {
  dv.paragraph("Нет достижений за эту неделю.");
} else {
  dv.table(["Достижение", "Дата", "XP"], rows);
}
```
