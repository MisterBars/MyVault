## XP фазы

```dataviewjs
const phaseTag = dv.current().phase_id;
const tasks = dv.pages()
  .where(x =>
    x.type === "task" &&
    x.status === "done" &&
    x.tags &&
    (Array.isArray(x.tags) ? x.tags.includes(phaseTag) : x.tags === phaseTag)
  ).array();

const earned = tasks.reduce((sum, t) => {
  if (t.xp) return sum + Number(t.xp);
  const m = { simple: 10, normal: 20, major: 40, boss: 100 };
  return sum + (m[t.task_type] || 20);
}, 0);

const total = tasks.length;
dv.paragraph(`✅ Выполнено задач: **${total}** | ⚡ Заработано XP: **${earned}**`);
```
