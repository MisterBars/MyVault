## Сводка прогресса
```dataviewjs
const project = dv.page('web3');
if (!project) {
  dv.paragraph("Проект web3 не найден.");
  return;
}

const pages = dv.pages()
  .where(p => p.project && p.project.path === project.file.path)
  .where(p => ["module","task","artifact","checkpoint","study-session"].includes(p.type))
  .array();

const byType = (t) => pages.filter(p => p.type === t);
const done = (arr) => arr.filter(p => p.status === "done");

const modules = byType("module");
const tasks = byType("task");
const artifacts = byType("artifact");
const checkpoints = byType("checkpoint");
const sessions = byType("study-session");

dv.table(
  ["Сущность", "Всего", "Готово"],
  [
    ["Модули", modules.length, done(modules).length],
    ["Задачи", tasks.length, done(tasks).length],
    ["Артефакты", artifacts.length, done(artifacts).length],
    ["Контрольные точки", checkpoints.length, done(checkpoints).length],
    ["Сессии", sessions.length, sessions.length]
  ]
);
```
