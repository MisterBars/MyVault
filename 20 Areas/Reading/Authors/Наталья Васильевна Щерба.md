function getCoverSrc(value) {
  if (!value) return null;

  try {
    const t = dv.func.typeof(value);

    if (t === "link" && value.path) {
      const file = app.metadataCache.getFirstLinkPathDest(value.path, "");
      if (!file) return null;
      return app.vault.getResourcePath(file);
    }

    if (t === "string") {
      const s = String(value).trim();
      if (!s) return null;

      if (/^https?:\/\//i.test(s)) {
        return s;
      }

      const file = app.metadataCache.getFirstLinkPathDest(s, "");
      if (file) {
        return app.vault.getResourcePath(file);
      }

      return s;
    }
  } catch (e) {
    return null;
  }

  return null;
}

const src = getCoverSrc(cover);

## Кратко
<p align="justify">Наталья Васильевна Щерба — украинская писательница в жанре подростковой фантастики и фэнтези. Родилась в городе Молодечно (Беларусь), детство провела в России, в городе Ярославле. Долгое время жила в Ивано-Франковске в Карпатах, а в 2020 году переехала в Словакию.</p>
<p align="justify">С детства очень любила читать — за школьные годы познакомилась с произведениями Купера, Дюма, Носова, Булычева и многих других авторов. Кроме литературы, увлекалась боевым единоборством ушу, которым занимается по сей день. Пробовала себя в различных профессиях: была корреспондентом, контент-менеджером, вела тренировки по ушу.</p>
<p align="justify">Первое публичное литературное произведение Натальи Щербы вышло в 2005 году — фантастический рассказ «На дне». В 2008 году вышел первый роман «Быть ведьмой», удостоенный ряда премий. В 2010 году завоевала гран-при конкурса «Новая детская книга» издательства «Росмэн» в номинации «Приключения и фэнтези» с книгой «Часовой ключ».</p>
<p align="justify">Самое популярное произведение — цикл «Часодеи», который был переведен и издан на Украине, в Польше, Чехии, Словакии, Венгрии и Болгарии.</p>

const a = dv.current();
const books = dv.pages('"Volodya/20 Area/Reading/Books"')
  .where(b => b.type === "book" && b.author && String(b.author.path || b.author) === a.file.path)
  .array();

function fmt(v, empty = "—") {
  if (v == null || v === "") return empty;
  if (Array.isArray(v)) return v.length ? v.join(", ") : empty;
  return String(v);
}

function num(v, def = 0) {
  const n = Number(v);
  return Number.isFinite(n) ? n : def;
}

function calcBookXP(book) {
  const pages = num(book.pages);
  if (pages <= 0) return 0;

  const current = Math.min(num(book.current_page), pages);
  const progress = current / pages;
  const baseXP = 100;
  const sizeCoef = num(book.size_coef, 1);
  const difficultyCoef = num(book.difficulty_coef, 1);
  const finishBonus = String(book.status || "") === "done" ? num(book.finish_bonus, 0) : 0;

  return Math.round(progress * baseXP * sizeCoef * difficultyCoef + finishBonus);
}

const doneBooks = books.filter(b => String(b.status || "") === "done");
const ratedBooks = books.filter(b => num(b.rating) > 0);
const avgRating = ratedBooks.length
  ? (ratedBooks.reduce((s, b) => s + num(b.rating), 0) / ratedBooks.length).toFixed(2)
  : "—";
const totalXP = doneBooks.reduce((s, b) => s + calcBookXP(b), 0);

dv.table(
  ["Параметр", "Значение"],
  [
    ["Страна", fmt(a.country)],
    ["Дата рождения", fmt(a.born)],
    ["Дата смерти", fmt(a.died)],
    ["Жанры", fmt(a.genres)],
    ["Книг в vault", books.length],
    ["Прочитано", doneBooks.length],
    ["Средний рейтинг", avgRating],
    ["Суммарный XP", totalXP]
  ]
);
