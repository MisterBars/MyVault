function calcXP(book) {
  const pages = num(book.pages);
  if (pages <= 0) return 0;
  const current = Math.min(num(book.current_page), pages);
  const progress = current / pages;
  const baseXP = 100;
  const sizeCoef = num(book.size_coef, 1);

    : 0;
  return Math.round(progress * baseXP * sizeCoef * difficultyCoef + finishBonus);
}

## О чём книга
<p align="justify">Гостем сэра Макса становится сэр Нумминорих Кута — и рассказывает о тубурских Мастерах снов, изамонских обычаях и Вечном Сне. Тубурская игра — старинный ритуал, правила которого никто не знает до конца, и участие в нём меняет всех, кто в него вступает. Финальная книга цикла подводит черту под эпохой в жизни мира Ехо.

<p align="justify">Историю рассказывает нюхач Нумминорих Кута. Однажды ему приходится в одиночку отправиться на поиски пропавшей сновидицы в Тулан — страну, где к грезам относятся серьезнее, чем к реальности. Поступив в ученики к Мастеру Совершенных Снов, герой узнает о ежегодном состязании за право получить волшебную шапку, в которой хранится сон, поглощающий спящего целиком и не дающий вернуться в явь. Но взамен головной убор дарует неограниченное могущество внутри сна.</p>
<p align="justify">Именно шапка Датчуха Вахурмаха виновата в том, что Ехо стал модным сновидческим курортом, и в том, что появился цикл «Сновидения Ехо». Вернее, виновата не сама шапка, а то, что с ней в финале романа сделал Нумминорих.</p>
