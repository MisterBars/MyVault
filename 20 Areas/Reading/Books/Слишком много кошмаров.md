function calcProgress(book) {
  const pages = num(book.pages);
  const current = Math.min(num(book.current_page), pages);
  if (pages <= 0) return 0;
  return Math.round((current / pages) * 100);
}

function calcXP(book) {
  const pages = num(book.pages);
  if (pages <= 0) return 0;
  const current = Math.min(num(book.current_page), pages);
  const progress = current / pages;
  const baseXP = 100;
  const sizeCoef = num(book.size_coef, 1);
  const difficultyCoef = num(book.difficulty_coef, 1);
  const finishBonus = String(book.status || "") === "done"
    ? num(book.finish_bonus, 0)
    : 0;
  return Math.round(progress * baseXP * sizeCoef * difficultyCoef + finishBonus);
}

## О чём книга
<p align="justify">Сэр Макс всё дальше погружается в загадочный мир сновидений — чем дальше, тем страшнее становятся его сны, и вскоре кошмары уже невозможно отличить от реальности. Новые встречи и расставания, новые чудовища — мистический Ехо преподносит всё новые сюрпризы. Книга усиливает атмосферу тревоги и непредсказуемости, сохраняя при этом фирменный юмор автора.</p>
<p align="justify">Макс продолжает вычислять застрявших сновидцев, с которыми происходят пугающие вещи. На ярких и жизнерадостных улицах Ехо некоторые переживают настолько мощные кошмары, что начинают терять рассудок.</p>
<p align="justify">Тем временем в Кодекс Хрембера — этакую конституцию Соединенного Королевства — вносят поправки. Теперь колдовать можно активнее, и в городе возрождаются старые традиции магической архитектуры. Что, если амбициозный проект молодого архитектурного гения — Дворец Ста Чудес — как-то связан с кошмарами сновидцев? Главный двигатель сюжета книги — пушкинский вопрос о том, совместимы ли гений и злодейство.</p>
